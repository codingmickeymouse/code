2) Create a Web Page and implement Java Script functions

Aim

To create a web tool that processes input text by converting it to lowercase, removing punctuation, and tokenizing it into individual words.

Algorithm

  1. Initialize Interface: Design the user interface with input (textarea), trigger (button), and output sections.
  2. Style Interface: Apply styles to improve layout, readability, and user experience.
  3. Define Processing Logic: Create a function that triggers on button click to handle text processing.
  4. Capture & Normalize Input: Retrieve user input and convert it to lowercase for uniformity.
  5. Clean & Tokenize Text: Remove punctuation using regular expressions and split the cleaned text into individual words (tokens).
  6. Output Results: Display the processed versions in their respective output sections.

Program

    Copied!
    
    <!DOCTYPE html>
<html>
<head>
  <title>Text Processor</title>
  <style>
    body { font-family: Arial, sans-serif; padding: 30px; max-width: 700px; margin: auto; }
    textarea { width: 100%; height: 100px; font-size: 16px; }
    button { margin-top: 10px; padding: 10px 20px; font-size: 16px; }
    .output { margin-top: 20px; }
    .output h3 { margin-bottom: 5px; }
    .output pre { background: #f5f5f5; padding: 10px; border-radius: 5px; }
  </style>
</head>
<body>
  <h1>Text Processing Tool</h1>
  <textarea id="inputText" placeholder="Enter your text here..."></textarea><br>
  <button onclick="processText()">Process</button>
  <div class="output">
    <h3>Lowercase Text:</h3>
    <pre id="lowercaseText"></pre>
    <h3>Text Without Punctuation:</h3>
    <pre id="noPunctuationText"></pre>
    <h3>Tokenized Words:</h3>
    <pre id="tokenizedText"></pre>
  </div>
  <script>
    function processText() {
      const text = document.getElementById("inputText").value;
      const lower = text.toLowerCase();
      document.getElementById("lowercaseText").textContent = lower;
      const noPunctuation = lower.replace(/[.,\/#!$%\\^&\\*;:{}=\\-_`~()]/g, "");
      document.getElementById("noPunctuationText").textContent = noPunctuation;
      const tokens = noPunctuation.split(/\\s+/).filter(w => w.length > 0);
      document.getElementById("tokenizedText").textContent = JSON.stringify(tokens, null, 2);
    }
  </script>
</body>
</html>

Output

Displays lowercase version, punctuation-free text, and tokenized words from input.

3) Build a Web Application using Sentimental Analysis API

Aim

To detect and display the sentiment (positive, negative, or neutral) of user-input text using a sentiment analysis API.

Algorithm

  1. Collect input from a text area.
  2. Send it to a backend via POST request.
  3. Use API to analyze sentiment.
  4. Classify result based on score.
  5. Send result back to frontend and display.

Program

    Copied!
    
    // Frontend (index.html)
<textarea id="userText"></textarea>
<button onclick="analyzeSentiment()">Analyze</button>
<div id="result"></div>

<script>
async function analyzeSentiment() {
  const text = document.getElementById("userText").value;
  const resultDiv = document.getElementById("result");
  if (!text) {
    resultDiv.textContent = "Please enter some text.";
    return;
  }
  resultDiv.textContent = "Analyzing...";
  try {
    const res = await fetch("http://localhost:3000/analyze", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ text })
    });
    const data = await res.json();
    resultDiv.innerHTML = `<strong>Sentiment:</strong> ${data.sentiment}<br><strong>Score:</strong> ${data.score}`;
  } catch (e) {
    resultDiv.textContent = "Error analyzing sentiment.";
  }
}
</script>

// Backend (server.js)
const express = require('express');
const bodyParser = require('body-parser');
const Sentiment = require('sentiment');
const app = express();
const sentiment = new Sentiment();
app.use(bodyParser.json());
app.post('/analyze', (req, res) => {
  const { text } = req.body;
  const result = sentiment.analyze(text);
  let label = 'neutral';
  if (result.score > 1) label = 'positive';
  else if (result.score < -1) label = 'negative';
  res.json({ sentiment: label, score: result.score });
});
app.listen(3000, () => console.log('Running...'));

Output

User enters text, clicks Analyze, and sees the sentiment and score returned by the API.

4) Build a Web Page that performs Linear Regression

Aim

To implement a linear regression model using TensorFlow.js that predicts house prices based on size.

Algorithm

  1. Load TensorFlow.js and Chart.js libraries.
  2. Define dataset and normalize it.
  3. Build and train the model.
  4. Make predictions.
  5. Visualize results using Chart.js.

Program

    Copied!
    
    <script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs"></script>
<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
<canvas id="chart" width="800" height="400"></canvas>
<script>
const sizes = [500, 700, 800, 1000, 1200, 1500];
const prices = [150, 200, 210, 250, 300, 330];
const xs = tf.tensor1d(sizes);
const ys = tf.tensor1d(prices);
const model = tf.sequential();
model.add(tf.layers.dense({ units: 1, inputShape: [1] }));
model.compile({ optimizer: 'sgd', loss: 'meanSquaredError' });

async function trainModel() {
  await model.fit(xs, ys, {
    epochs: 300,
    callbacks: {
      onEpochEnd: (epoch, logs) => {
        if (epoch % 50 === 0) console.log(`Epoch ${epoch}: loss = ${logs.loss}`);
      }
    }
  });
  document.getElementById("status").textContent = "Training Complete";
  plot();
}

async function plot() {
  const chartData = {
    datasets: [
      { label: "Original", data: sizes.map((s, i) => ({ x: s, y: prices[i] })), backgroundColor: "blue" },
      { label: "Regression Line", data: [], type: "line", borderColor: "red", fill: false }
    ]
  };
  const lineX = [400, 1600];
  const preds = model.predict(tf.tensor1d(lineX));
  const predYs = await preds.array();
  chartData.datasets[1].data = [{ x: lineX[0], y: predYs[0] }, { x: lineX[1], y: predYs[1] }];
  new Chart(document.getElementById("chart"), { type: "scatter", data: chartData });
}
trainModel();
</script>

Output

A chart displaying actual data points and a predicted regression line based on trained model.

5) Adobe Sensei Simulation (Content-Aware Fill)

Aim

To simulate Adobe Photoshop’s Content-Aware Fill feature using JavaScript and canvas.

Algorithm

  1. Load image into canvas.
  2. Select an area to be filled.
  3. Sample the selection and fill with average color.

Program

    Copied!
    
    <input type="file" id="upload" />
<canvas id="canvas"></canvas>
<button id="fillButton">Fill Selected Area</button>

<script>
const upload = document.getElementById('upload');
const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
const fillButton = document.getElementById('fillButton');
let img = new Image();
let startX, startY, endX, endY, isDrawing = false;

upload.onchange = e => {
  const reader = new FileReader();
  reader.onload = ev => {
    img.onload = () => {
      canvas.width = img.width;
      canvas.height = img.height;
      ctx.drawImage(img, 0, 0);
    };
    img.src = ev.target.result;
  };
  reader.readAsDataURL(e.target.files[0]);
};

canvas.onmousedown = e => {
  const rect = canvas.getBoundingClientRect();
  startX = e.clientX - rect.left;
  startY = e.clientY - rect.top;
  isDrawing = true;
};

canvas.onmouseup = e => {
  if (!isDrawing) return;
  const rect = canvas.getBoundingClientRect();
  endX = e.clientX - rect.left;
  endY = e.clientY - rect.top;
  isDrawing = false;
  ctx.strokeStyle = 'red';
  ctx.lineWidth = 2;
  ctx.strokeRect(startX, startY, endX - startX, endY - startY);
};

fillButton.onclick = () => {
  if ([startX, startY, endX, endY].some(v => v === undefined)) return alert("Select an area first.");
  const width = endX - startX;
  const height = endY - startY;
  const data = ctx.getImageData(startX, startY, width, height).data;
  let r = 0, g = 0, b = 0;
  for (let i = 0; i < data.length; i += 4) {
    r += data[i]; g += data[i + 1]; b += data[i + 2];
  }
  const pixels = data.length / 4;
  r = Math.floor(r / pixels); g = Math.floor(g / pixels); b = Math.floor(b / pixels);
  ctx.fillStyle = `rgb(${r},${g},${b})`;
  ctx.fillRect(startX, startY, width, height);
};
</script>

Output

User uploads an image, selects an area, and fills it with average sampled color.
